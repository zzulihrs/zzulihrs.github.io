5970. 参加会议的最多员工数

leetcode第274周赛D题
还是挺有意思
思路：
两种情况
1）A, B互相喜欢，两个人的追随者都可以放上，找出所有符合条件的情况
2）最大环，只能放一个
取两种情况的最大值

#include <bits/stdc++.h>

#define db1(x) cout << #x << " = " << x << endl

#define gg cout << "---------QAQ--------\n"
#define MEM(a, b) memset(a, b, sizeof(a))

#define IOS ios::sync_with_stdio(false),cin.tie(NULL);

using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<pii, int> piii;
const int N = 2e5+10, M = 65;
const int mod = 1713222373;

class Solution {
public:
	#define maxsize 100010
	#define pb push_back
	#define fi first
	#define se second
	#define SZ(x) (int)x.size()
	#define ALL(x) x.begin(), x.end()
	/*
	两种情况
	1）A, B互相喜欢，两个人的追随者都可以放上，找出所有符合条件的情况
	2）最大环，只能放一个
	取两种情况的最大值
	*/

	vector<int> e[maxsize], re[maxsize];
	int d[maxsize];

    int maximumInvitations(vector<int>& a) {
    	int n = SZ(a);

  		for(int i = 0; i < SZ(a); ++i)
  		{
  			e[i].pb(a[i]);
  			re[a[i]].pb(i);
  			d[a[i]]++;
  		}
  		queue<int> q; 
  		vector<int> dp(n, 1);
  		for(int i = 0; i < n; ++i) 
  			if(!d[i]) q.push(i);
  		while(SZ(q)) {
  			int u = q.front(); q.pop();
  			dp[a[u]] = max(dp[a[u]], dp[u]+1);
  			if(--d[a[u]] == 0) q.push(a[u]);
  		}
  		int ans = 0;
  		for(int i = 0; i < n; ++i)
  			if(a[a[i]] == i) //两个之间互相喜欢，形成一个集合
  				ans += dp[i]+dp[a[i]];
 		ans /= 2; //计算了A, B和B, A两种情况，去重

  		//求最大环
  		vector<int> vis(n, 0);
  		for(int i = 0; i < n; ++i) {
  			if(vis[i]) continue;
  			vector<int> b;
  			int x = i;
  			while(!vis[x]) { 
  				vis[x] = 1;
  				b.pb(x);
  				x = a[x];
  			}
  			for(int j = 0; j < SZ(b); ++j)
  				if(b[j] == x)  //x是环上的点，也是与链的交点
  				ans = max(ans, SZ(b)-j);
  		}


  		return ans;	      
    }
}T;

void sol()
{
	vector<int> a{3,0,1,4,1};
	db1(T.maximumInvitations(a));
}
signed main()
{	
	//IOS
    //int _ = 1;	cin>>_;	while (_--)
        sol();
}
